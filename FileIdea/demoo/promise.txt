promise               lời hứa sẽ trả kết quả về bất kể thành công hay thất bại
                      promise chạy bất đồng bộ

new Promise(function (res, rej){        res: trả kết quả khi thành công
  //....code                            rej: trả kết quẩ khi thất bại
})


.then(function(data){                   .then để bắt kết kết quả nếu promise thành công
  //...                                 data là kết quả trả về khi thành công
})                                      then return ra 1 promise mới

.catch(function(err){                   .catch để bắt kết kết quả nếu promise thất bại
  //...                                 err là lỗi trả về khi thất bại
})                                      then return ra 1 promise mới
